1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "restarters.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "restarters.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "restarters.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "restarters.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. External API Keys Configuration:
   Secrets are enabled for external API keys (MAPBOX_TOKEN, GOOGLE_API_CONSOLE_KEY).

   {{- if .Values.secrets.createSecret }}
   A secret named "{{ .Values.secrets.secretName }}" has been created with placeholder values.
   Update it with your actual API keys:

   kubectl create secret generic {{ .Values.secrets.secretName }} \
     --from-literal={{ .Values.secrets.keys.mapboxToken }}="your-actual-mapbox-token" \
     --from-literal={{ .Values.secrets.keys.googleApiKey }}="your-actual-google-api-key" \
     --dry-run=client -o yaml | kubectl apply -f -
   {{- else }}
   Make sure the secret "{{ .Values.secrets.secretName }}" exists with the following keys:
   - {{ .Values.secrets.keys.mapboxToken }}
   - {{ .Values.secrets.keys.googleApiKey }}

   You can create it manually:
   kubectl create secret generic {{ .Values.secrets.secretName }} \
     --from-literal={{ .Values.secrets.keys.mapboxToken }}="your-mapbox-token" \
     --from-literal={{ .Values.secrets.keys.googleApiKey }}="your-google-api-key"

   Or use ExternalSecrets to sync from your secret management system.
   {{- end }}
{{- else }}

2. External API Keys:
   API keys are configured via values.yaml. To use Kubernetes secrets instead:
   - Set secrets.enabled=true
   - Optionally set secrets.createSecret=true for testing
   - Or use ExternalSecrets for production secret management
{{- end }}
